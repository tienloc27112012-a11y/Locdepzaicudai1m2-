local function quantumDesync()
    local Backpack = LocalPlayer:WaitForChild("Backpack")
    local Character = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local Humanoid = Character:WaitForChild("Humanoid")
    local QuantumCloner = Backpack:FindFirstChild("Quantum Cloner") or Character:FindFirstChild("Quantum Cloner")
    
    if not QuantumCloner then
        return false
    end
    
    if Character:FindFirstChild(QuantumCloner.Name) ~= QuantumCloner then
        Humanoid:EquipTool(QuantumCloner)
    end
    
    local Packages = ReplicatedStorage:WaitForChild("Packages", 5)
    local Net = Packages and Packages:FindFirstChild("Net")
    
    if not Net then
        return false
    end
    
    local UseItem = Net:FindFirstChild("RE/UseItem")
    local OnTeleport = Net:FindFirstChild("RE/QuantumCloner/OnTeleport")
    
    if UseItem and OnTeleport then
        UseItem:FireServer()
        OnTeleport:FireServer()
        
        local Notify = ReplicatedStorage.Packages.Net["RE/NotificationService/Notify"]
        for _ = 1, 3 do
            firesignal(Notify.OnClientEvent, "<font color=\"#db1d1d\">Porxiai Hub On Top!!!</font>", 5, "Sounds.Sfx.Success")
        end
        
        for flagName, flagValue in pairs(DesyncFlags) do
            pcall(function()
                setfflag(tostring(flagName), tostring(flagValue))
            end)
        end
        return true
    end
    return false
end

local function noToolDesync()
    local sequence = {
        teleportIndices = {2, 4},
        sequence = {
            {flag = "WorldStepsOffsetAdjustRate", value = "-1", delay = 0},
            {flag = "WorldStepsOffsetAdjustRate", value = "60", delay = 12.57},
            {flag = "WorldStepsOffsetAdjustRate", value = "-9999999999", delay = 52.14},
            {flag = "WorldStepsOffsetAdjustRate", value = "-9999999999", delay = 66.15},
            {flag = "WorldStepsOffsetAdjustRate", value = "-1", delay = 1447.19},
        }
    }
    
    task.spawn(function()
        for flagName, flagValue in pairs(DesyncFlags) do
            pcall(function()
                setfflag(tostring(flagName), tostring(flagValue))
            end)
        end
        
        for i, entry in ipairs(sequence.sequence) do
            if entry.delay > 0 then 
                task.wait(entry.delay / 1000) 
            end
            
            pcall(function() 
                setfflag(entry.flag, entry.value) 
            end)
            
            if table.find(sequence.teleportIndices, i) then
                local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
                local hrp = char:WaitForChild("HumanoidRootPart")
                if hrp then
                    local max = 20
                    hrp.CFrame = hrp.CFrame * CFrame.new(
                        math.random(-max, max),
                        0,
                        math.random(-max, max)
                    )
                end
            end
        end
    end)
end

local function toggleMinimize()
    isMinimized = not isMinimized
    MinimizeButton.Text = isMinimized and "+" or "-"
    ContentFrame.Visible = not isMinimized
    
    local targetSize = isMinimized and minimizedSize or originalSize
    TweenService:Create(MainFrame, TweenInfo.new(0.2), {
        Size = targetSize
    }):Play()
    
    if writefile then
        pcall(function()
            writefile("PorxiaiHub_Minimized.txt", tostring(isMinimized))
        end)
    end
end

local function setupButtonHover(button)
    local originalColor = button.BackgroundColor3
    
    button.MouseEnter:Connect(function()
        button.BackgroundColor3 = originalColor:Lerp(Color3.fromRGB(255, 255, 255), 0.1)
    end)
    
    button.MouseLeave:Connect(function()
        button.BackgroundColor3 = originalColor
    end)
end

local dragging = false
local dragStart, startPos

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
    end
end)

MainFrame.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(
            startPos.X.Scale, 
            startPos.X.Offset + delta.X,
            startPos.Y.Scale, 
            startPos.Y.Offset + delta.Y
        )
    end
end)
